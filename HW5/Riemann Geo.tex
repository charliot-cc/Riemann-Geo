\noindent\textbf{7-5}
\begin{proof}
    To calculate $\rr{kli}{j}$, we look at $R(E_k,E_l)E_i$. By definition, we have
    \[R(E_k,E_l)E_i = \n_{E_k}\n_{E_l}E_i - \n_{E_l}\n_{E_k}E_i - \n_{[E_k,E_l]}E_i.\]
    Where
    \[\n_{E_k}\n_{E_l}E_i = \n_{E_k}(\w{i}{m}(E_l)E_m) = (E_k(\w{i}{m}(E_l)) + \w{i}{n}(E_l)\w{n}{k}(E_k))E_m.\]
    \newcommand{\aaa}[4]{E_#1(\w{#3}{#4}(E_#2)) + \w{#3}{m}(E_#2)\w{m}{#1}(E_#1)}
    Similarly, we have $\n_{E_l}\n_{E_k}E_i = (E_l(\w{i}{m}(E_k)) + \w{i}{m}(E_k)\w{m}{l}(E_l))E_m.$. Besides,
    \[\n_{[E_k,E_l]}E_i = \w{i}{m}([E_k,E_l])E_m.\]
    Therefore, we have 
    \[\rr{kli}{j} = \aaa{k}{l}{i}{j} - \aaa{l}{k}{i}{j} - \w{i}{j}([E_k,E_l]).\]
    For any frame elements $E_j,\,E_k$, we have
    \begin{align*}
        \W{i}{j}(E_k,E_l) &= \frac{1}{2}\rr{mni}{j} \eps^m\wedge\eps^n (E_k,E_l)\\
        &= \rr{kli}{j}\\ &= \aaa{k}{l}{i}{j} - \aaa{l}{k}{i}{j} - \w{i}{j}([E_k,E_l]).
    \end{align*}
    For the LHS,
    \[d\w{i}{j}(E_k,E_l) = E_k(\w{i}{j}(E_l)) - E_l(\w{i}{j}(E_k)) - \w{i}{j}([E_k,E_l]).\]
    and
    \[\w{i}{m}\wedge\w{m}{j}(E_k,E_l) = \w{i}{m}(E_k)\w{m}{j}(E_l) - \w{i}{m}(E_l)\w{m}{j}(E_k).\]
    By simplying the RHS, we have
    \[\W{i}{j}(E_k,E_l) = d\w{i}{j}(E_k,E_l) + \w{i}{m}\wedge\w{m}{j}(E_k,E_l).\]   
    Therefore, we have $\W{i}{j} = d\w{i}{j} + \w{i}{m}\wedge\w{m}{j}$.
\end{proof}

\noindent\textbf{7-14}
\newcommand{\trm}[4]{\widetilde{Rm}(\tilde#1,\tilde#2,\tilde#3,\tilde#4)}
\newcommand{\tr}[3]{\widetilde{R}(\tilde#1,\tilde#2,\tilde#3)}
\newcommand{\mt}[1]{\langle #1 \rangle}
\newcommand{\tx}{\tilde X}
\newcommand{\ty}{\tilde Y}
\newcommand{\tz}{\tilde Z}

\newcommand{\wt}{\widetilde}
\newcommand{\hlf}{\frac{1}{2}}
\newcommand{\qd}{\frac{1}{4}}
\begin{proof}
    To calculate $\trm{W}{X}{Y}{Z}$, we compute
    \[\tr{W}{X}{Y}=\tn_{\tw}\tn_{\tx}\ty - \tn_{\tx}\tn_{\tw}\ty - \tn_{[\tw,\tx]}\ty.\]
    We first look at the term with two $\n$'s,
    \begin{align*}
        \tn_{\tw}\tn_{\tx}\ty &= \tn_{\tw}(\wt{\n_X Y} + \hlf [\tx,\ty]^V)\quad\text{(By prob 5-6)}\\
        &= \wt{\n_W\n_X Y} + \hlf [\tw,\wt{\n_X Y}]^V + \hlf \tn_{\tw}[\tx,\ty]^V.
    \end{align*}
    After inner product with $\tz$, we have
    \[\mt{\tn_{\tw}\tn_{\tx}\ty,\tz} = \mt{\wt{\n_W\n_X Y},\tz} + \hlf\mt{\tn_{\tw} [\tx,\ty]^V,\tz}.\]
    By Prob 5-6, the first term is $\mt{\n_W\n_X Y,Z}\circ\pi$. For the second term, we use metric compatibility
    \begin{align*}
        \hlf\mt{\tn_{\tw} [\tx,\ty]^V,\tz} &= \hlf\left(\tw\mt{[\tx,\ty]^V\tz} - \mt{[\tx,\ty]^V,\tn_{\tw}\tz}\right)\\
        &= 0 - \qd\mt{[\tx,\ty]^V,[\tw,\tz]^V}
    \end{align*} 
    To conclude, we have $\mt{\tn_{\tw}\tn_{\tx}\ty,\tz} = \mt{\n_W\n_X Y,Z}\circ\pi -\qd\mt{[\tx,\ty]^V,[\tw,\tz]^V}$.
    \newcommand{\cm}[4]{\mt{[#1,#2]^V,[#3,#4]^V}}
    By symmetric argument, we have $\mt{\tn_{\tx}\tn_{\tw}\ty,\tz} = \mt{\n_X\n_W Y,Z}\circ\pi -\qd\cm{W}{Y}{X}{Z}$.

    For the last term, 
    \[\mt{\tn_{[\tw,\tx]}\ty,\tz} = \mt{\tn_{[\tw,\tx]^H}\ty,\tz}+\mt{\tn_{[\tw,\tx]^V}\ty,\tz}.\]
    For the first term, we have
    \[\mt{\tn_{[\tw,\tx]^H}\ty,\tz} = \mt{\tn_{\wt{[W,X]}}\ty,\tz} = \mt{\n_{[W,X]} Y,Z}\circ\pi.\]
    For the second term, we use torsion free property,
    \[\tn_{[\tw,\tx]^V}\ty = [[\tw,\tx]^V,\ty] + \tn_{\ty} [\tw,\tx]^V.\]
    By prob 5-6, $[[\tw,\tx]^V,\ty]$ is vertical, and $\mt{\tn_{\ty} [\tw,\tx]^V,\tz}$ has the same structure as we have calculated before.
    Therefore, we have $\mt{\tn_{[\tw,\tx]}\ty,\tz} = \mt{\n_{[W,X]} Y,Z}\circ\pi - \hlf\cm{W}{X}{Y}{Z}$.

    To conclude, we have
    \begin{align*}
        \trm{W}{X}{Y}{Z} &= \mt{\tn_{\tw}\tn_{\tx}\ty,\tz} - \mt{\tn_{\tx}\tn_{\tw}\ty,\tz} - \mt{\tn_{[\tw,\tx]}\ty,\tz}\\
                        &= \mt{\n_W\n_X Y,Z}\circ\pi -\qd\cm{X}{Y}{W}{Z} - \mt{\n_X\n_W Y,Z}\circ\pi \\ &\quad +\qd\cm{W}{Y}{X}{Z} - \mt{\n_{[W,X]} Y,Z}\circ\pi + \hlf\cm{W}{X}{Y}{Z}\\
                        &= R(W,X,Y,Z)\circ\pi - \qd\cm{X}{Y}{W}{Z} + \qd\cm{W}{Y}{X}{Z} + \hlf\cm{W}{X}{Y}{Z}.
                        \end{align*}
\end{proof}
